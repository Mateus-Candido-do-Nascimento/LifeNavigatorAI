# 4. A função principal:
jsxasync function enviarMensagem() {
    if (!input.trim() || carregando) return
    // guarda contra envio vazio ou duplo clique

    const mensagemUsuario = { role: "user", content: input }

    setMensagens((prev) => [...prev, mensagemUsuario])
    // prev = mensagens anteriores
    // ...prev = spread — copia tudo
    // adiciona a nova no final
    // React atualiza a tela automaticamente

    setInput("")        // limpa o campo
    setCarregando(true) // ativa o "Analisando..."

    try {
        const response = await axios.post("http://localhost:8000/api/chat/", {
            mensagem: input,
            session_id: SESSION_ID,
        })
        // await = espera a resposta do backend
        // igual ao requests.post() do Python

        const mensagemBot = {
            role: "assistant",
            content: response.data.resposta,
        }
        setMensagens((prev) => [...prev, mensagemBot])
        // adiciona resposta do bot na tela

    } catch (erro) {
        // se o backend cair, mostra mensagem de erro
        setMensagens((prev) => [...prev, {
            role: "assistant",
            content: "Erro ao conectar com o servidor.",
        }])
    } finally {
        setCarregando(false) // sempre desativa o carregando
    }
}

# 5. Enter pra enviar:
jsxfunction handleKeyDown(e) {
    if (e.key === "Enter" && !e.shiftKey) {
        e.preventDefault()  // impede quebra de linha
        enviarMensagem()
    }
    // Shift+Enter → quebra linha
    // Enter sozinho → envia
}
